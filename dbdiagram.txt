Table Player {
  name varchar [pk]
  score integer
}

Table Game {
  id integer [pk, increment]
  player_name varchar [ref: > Player.name]
}

Table Clue {
  id integer [pk, increment]
  question text
  answer text
}

Table GameClues {
  game_id integer [ref: > Game.id]
  clue_id integer [ref: > Clue.id]
}
